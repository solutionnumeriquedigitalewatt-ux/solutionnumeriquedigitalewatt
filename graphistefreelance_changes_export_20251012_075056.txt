--- init-and-commit.ps1 ---
# Init and commit script for graphistefreelance
# Usage: Open PowerShell and run this script after installing Git

$repo = Split-Path -Parent $MyInvocation.MyCommand.Definition
Write-Output "Running in: $repo"

# Check git availability
try {
    git --version | Out-Null
} catch {
    Write-Error "Git not found. Please install Git for Windows from https://git-scm.com/download/win and re-run this script."
    exit 1
}

# Initialize repository if not already
if (-not (Test-Path (Join-Path $repo ".git"))) {
    git -C $repo init
    Write-Output "Initialized new git repo in $repo"
} else {
    Write-Output "Git repo already initialized"
}

# Set local user config if missing
$hasName = git -C $repo config user.name
if (-not $hasName) {
    $name = Read-Host "Enter git user.name to set locally (e.g. Your Name)"
    if ($name) { git -C $repo config user.name "$name" }
}
$hasEmail = git -C $repo config user.email
if (-not $hasEmail) {
    $email = Read-Host "Enter git user.email to set locally (e.g. you@example.com)"
    if ($email) { git -C $repo config user.email "$email" }
}

# Add and commit
git -C $repo add -A
$defaultMessage = "Normalize styles: move inline styles to CSS, clean HTML, add utility classes"
$message = Read-Host "Commit message (leave empty for default)"
if (-not $message) { $message = $defaultMessage }

# Commit (if there are changes)
$changes = git -C $repo status --porcelain
if ($changes) {
    git -C $repo commit -m "$message"
    Write-Output "Committed changes with message: $message"
    git -C $repo --no-pager log -n 5 --oneline
} else {
    Write-Output "No changes to commit"
}
